  • The root node is an empty node.
  • Every node of the tree represents a character (exclude the root node)
  • Every node has 0 to 26 childrens.
  • A path from the root to the leaf is a string in the dataset.
  • A path can represent multiple strings.

class Trie {

}

public void insert(String key) {

TrieNode currNode = root;

for (char ch : key.toCharArray()) {

int index = charToIndex(ch);

if (currNode.children[index] == null) {

currNode.children[index] = new TrieNode();

}

currNode = currNode.children[index];

}

currNode.isEndOfWord = true;

}

public boolean search(String key) {

TrieNode currNode = root;

for (char ch : key.toCharArray()) {

int index = charToIndex(ch);

if (currNode.children[index] == null) {

return false;

}

currNode = currNode.children[index];

}

return currNode.isEndOfWord;

}
